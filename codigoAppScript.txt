function doPost(e) {

  const sheet =
    SpreadsheetApp
      .getActiveSpreadsheet()
      .getActiveSheet();

  const d =
    JSON.parse(e.postData.contents);

  let acpmTerpel = 0;
  let acpmEfectivo = 0;
  let peajes = 0;
  let urea = 0;
  let parqueadero = 0;
  let cargue = 0;
  let descargue = 0;
  let comision = 0;
  let lavado = 0;
  let mantenimiento = 0;     // ← NUEVO acumulador
  let otros = 0;

  let motivo = [];
  let descripcionMtto = [];  // ← NUEVO array para las notas de mantenimiento

  (d.gastos || []).forEach(g => {

    const concepto =
      (g.concepto || "")
        .toUpperCase()
        .trim();

    const valor =
      Number(g.valor) || 0;

   if (concepto.includes("ACPM TERPEL")) {

      acpmTerpel += valor;

    } else if (concepto.includes("ACPM EFECTIVO")) {

      acpmEfectivo += valor;

    } else if (concepto.includes("PEAJE")) {

      peajes += valor;

    } else if (concepto.includes("UREA")) {

      urea += valor;

    } else if (concepto.includes("PARQUEADERO")) {

      parqueadero += valor;

    } else if (
      concepto === "CARGUE M/CIA" ||
      concepto === "CARROZADA"
    ) {

      cargue += valor;

    } else if (
      concepto === "DESCARGUE M/CIA" ||
      concepto === "DESCARROZADA"
    ) {

      descargue += valor;

    } else if (
      concepto.includes("COMISION")
    ) {

      comision += valor;

    } else if (
      concepto.includes("LAVADO")
    ) {

      lavado += valor;

    } else if (
      concepto.includes("MANTENIMIENTO")   // ← NUEVO bloque
    ) {

      mantenimiento += valor;

      if (g.nota) {
        descripcionMtto.push(g.nota);
      }

    } else if (
      concepto.includes("EVENTUAL")
    ) {

      otros += valor;

      if (g.nota) {
        motivo.push(g.nota);
      }

    }

  });

  const anticipos =
    Number(d.anticipo1) || 0;

  const flete =
    Number(d.flete) || 0;

  const ruta =
    (d.origen || "").toUpperCase() +
    " - " +
    (d.destino || "").toUpperCase();

  const motivoTexto =
    motivo.join(" , ").toUpperCase();

  const descripcionMttoTexto =          // ← NUEVO texto unido
    descripcionMtto.join(" , ").toUpperCase();

  sheet.appendRow([

    d.fecha,                    // FECHA
    d.mnf,                      // MNF
    d.placa,                    // PLACA
    ruta,                       // RUTA
    "",                         // MES CIERRE
    d.conductor.toUpperCase(),  // CONDUCTOR

    d.km_inicio,                // KM SALIDA
    "",                         // KM LLEGADA
    "",                         // KMS RECORRIDOS

    flete,                      // FLETE
    anticipos,                  // ANTICIPOS

    peajes,                     // PEAJES
    d.galones_efectivo,         // GALONES EFECTIVO

    acpmEfectivo,               // PRECIO ACPM EFECTIVO
    "",                         // KM/GL
    "",                         // LITROS

    urea,                       // UREA
    parqueadero,                // PARQUEADERO

    cargue,                     // CARGUE
    descargue,                  // DESCARGUE

    comision,                   // COMISION
    "",                         // BANCOS

    lavado,                     // LAVADO
    mantenimiento,              // M/TTO
    descripcionMttoTexto,       // DESCRIPCION

    otros,                      // OTROS
    motivoTexto,                // MOTIVO

    "",                         // % CONDUCTOR
    d.galones_terpel,           // GLN TERPEL
    acpmTerpel,                 // PRECIO TERPEL

    "",                         // TOTAL GASTOS
    "",                         // UTILIDAD BRUTA

    "",                         // RENT VIAJE
    "",                         // SUELDO CONDUCTOR

    d.empresa,                  // EMPRESA

    "",                         // SALDO

    "",                         // FECHA DE PAGO
    "",                         // VR CANCELADO
    "",                         // ESTADO PAGO
    "",                         // FACT. ELECT.
    "",                         // PROVEEDOR
    "",                         // PRODUCTO
    "",                         // VALOR

    ""                          // UTILIDAD NETA

  ]);

  return ContentService
    .createTextOutput(
      JSON.stringify({
        status: "ok"
      })
    )
    .setMimeType(
      ContentService.MimeType.JSON
    );
}
